# wsitools v0.17 — ScaledStrips Wiring Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Rewire `convert --to dzi|szi` to render DZI pyramids through opentile-go v0.26's `ScaledStrips` iterator instead of per-tile `ReadRegionScaled` calls. Unblocks the skipped NDPI integration tests by making striped-source DZI rendering several times faster.

**Architecture:** Per-level dispatch in `emitDZIPyramid`: levels where row count is `>1` get a strip iterator with a 3-strip rolling buffer (`prev`/`cur`/`next`) for overlap handling; the existing per-tile `ReadRegionScaled` path is kept as a fallback for the coarsest levels where iterator overhead wouldn't amortize. No new CLI flags; existing `--workers` controls decode parallelism.

**Tech Stack:** Go 1.23, opentile-go v0.26 (`*Slide.ScaledStrips`, `*StripIterator`), `decoder.Image` RGB pixel buffers, stdlib `image/jpeg` + `image/png`. No cgo on this change path.

**Branch:** `feat/v17-scaledstrips-dzi`.

**Spec:** `docs/superpowers/specs/2026-05-26-v0.17-scaledstrips-wiring-design.md`.

// emitLevelPerTile renders one DZI level by calling
// slide.ReadRegionScaled for each tile region (the v0.16 path).
// Kept as a fallback for levels where ScaledStrips' setup overhead
// wouldn't amortize (rows<=1; coarsest levels).
func emitLevelPerTile(ctx context.Context, slide *opentile.Slide, w dziTileSink, cfg dzi.Config,
	lvl, cols, rows, quality int) error {
	for row := 0; row < rows; row++ {
		for col := 0; col < cols; col++ {
			if err := ctx.Err(); err != nil {
				return err
			}
			body, err := renderDZITilePerTile(slide, cfg, lvl, col, row, quality)
			if err != nil {
				return fmt.Errorf("level %d (%d,%d): %w", lvl, col, row, err)
			}
			if err := w.WriteTile(lvl, col, row, body); err != nil {
				return err
			}
		}
	}
	return nil
}

// makeStrip builds a synthetic decoder.Image (RGB) of size w*h
// where each pixel encodes (R, G, B) = (rowMod, colMod, stripID).
// rowMod/colMod are y/x truncated to byte for predictable patterns.
func makeStrip(w, h int, stripID byte) *decoder.Image {
	pix := make([]byte, w*h*3)
	for y := 0; y < h; y++ {
		for x := 0; x < w; x++ {
			i := y*w*3 + x*3
			pix[i+0] = byte(y)
			pix[i+1] = byte(x)
			pix[i+2] = stripID
		}
	}
	return &decoder.Image{
		Width: w, Height: h, Stride: w * 3, Pix: pix,
		Format: decoder.PixelFormatRGB,
	}
}

```go
// encodeDZITileFromStrips assembles one DZI tile from up to three
// strip images (prev/cur/next) and encodes per cfg.Format.
//
// Strip layout: strip r covers DZI level rows [r*TileSize,
// (r+1)*TileSize); strip width = lw (the full level width). The
// content rows of DZI tile (col, row) are sourced directly from
// the corresponding strip; top/bottom overlap rows come from
// prev/next; left/right overlap comes from neighboring columns in
// the same strip.
func encodeDZITileFromStrips(
	prev, cur, next *decoder.Image,
	cfg dzi.Config,
	lvl, lw, lh, col, row, quality int,
) ([]byte, error) {
	cw, ch := dzi.EdgeTileDims(lw, lh, cfg.TileSize, col, row)

	// Overlap on each side (suppressed at level edges).
	leftOv := cfg.Overlap
	if col == 0 {
		leftOv = 0
	}
	topOv := cfg.Overlap
	if row == 0 || prev == nil {
		topOv = 0
	}
	rightOv := cfg.Overlap
	if col*cfg.TileSize+cw >= lw {
		rightOv = 0
	}
	bottomOv := cfg.Overlap
	if row*cfg.TileSize+ch >= lh || next == nil {
		bottomOv = 0
	}

	outW := cw + leftOv + rightOv
	outH := ch + topOv + bottomOv

	dst := image.NewRGBA(image.Rect(0, 0, outW, outH))

	// Source x-range in strip coordinates: same for prev/cur/next.
	srcX0 := col*cfg.TileSize - leftOv

	// Top band from prev: take prev's bottom topOv rows.
	if topOv > 0 {
		prevStart := prev.Height - topOv
		copyStripBandRGBToRGBA(dst, 0, prev, prevStart, srcX0, outW, topOv)
	}

	// Middle band from cur: cur's full content rows.
	copyStripBandRGBToRGBA(dst, topOv, cur, 0, srcX0, outW, ch)

	// Bottom band from next: top bottomOv rows.
	if bottomOv > 0 {
		copyStripBandRGBToRGBA(dst, topOv+ch, next, 0, srcX0, outW, bottomOv)
	}

	var out bytes.Buffer
	switch cfg.Format {
	case "jpeg":
		if err := jpeg.Encode(&out, dst, &jpeg.Options{Quality: quality}); err != nil {
			return nil, err
		}
	case "png":
		if err := png.Encode(&out, dst); err != nil {
			return nil, err
		}
	default:
		return nil, fmt.Errorf("unsupported --dzi-format %q", cfg.Format)
	}
	return out.Bytes(), nil
}

// copyStripBandRGBToRGBA copies a horizontal band of pixels from an
// RGB decoder.Image into a destination image.RGBA. The destination
// band starts at dstY for h rows; the source band is rows
// [srcY, srcY+h) and columns [srcX, srcX+w). srcX may be negative
// or extend past the strip width — out-of-range pixels are
// transparent-white-filled to handle level-edge clipping; the DZI
// overlap-suppression rules ensure this only happens at unsupressed
// edges, which is a programming error, but we tolerate it instead
// of panicking.
func copyStripBandRGBToRGBA(dst *image.RGBA, dstY int, src *decoder.Image, srcY, srcX, w, h int) {
	for r := 0; r < h; r++ {
		dstRow := dst.Pix[(dstY+r)*dst.Stride : (dstY+r)*dst.Stride+w*4]
		srcRowStart := (srcY + r) * src.Stride
		for c := 0; c < w; c++ {
			sx := srcX + c
			var rr, gg, bb byte
			if sx >= 0 && sx < src.Width {
				si := srcRowStart + sx*3
				rr = src.Pix[si+0]
				gg = src.Pix[si+1]
				bb = src.Pix[si+2]
			} else {
				rr, gg, bb = 0xFF, 0xFF, 0xFF
			}
			dstRow[c*4+0] = rr
			dstRow[c*4+1] = gg
			dstRow[c*4+2] = bb
			dstRow[c*4+3] = 0xFF
		}
	}
}
```

```go
// emitLevelViaStrips renders one DZI level by opening a
// *StripIterator with stripHeight == cfg.TileSize and consuming
// strips through a 3-strip rolling buffer (prev/cur/next) so each
// DZI tile can pull its top/bottom overlap rows from the adjacent
// strips. Use for levels where rows > 1; the per-tile fallback
// covers smaller levels where iterator setup wouldn't amortize.
func emitLevelViaStrips(ctx context.Context, slide *opentile.Slide, w dziTileSink, cfg dzi.Config,
	lvl, lw, lh, cols, rows, workers, quality int) error {

	stripOpts := []opentile.StripOption{
		opentile.WithStripContext(ctx),
	}
	if workers > 0 {
		stripOpts = append(stripOpts, opentile.WithStripWorkers(workers))
	}

	it := slide.ScaledStrips(
		image.Rect(0, 0, cfg.Width, cfg.Height),
		image.Point{X: lw, Y: lh},
		cfg.TileSize,
		stripOpts...,
	)
	defer it.Close()

	cur, err := it.Next()
	if err != nil {
		return fmt.Errorf("level %d strip 0: %w", lvl, err)
	}
	var prev *decoder.Image

	for row := 0; row < rows; row++ {
		var next *decoder.Image
		if row+1 < rows {
			next, err = it.Next()
			if err != nil {
				return fmt.Errorf("level %d strip %d: %w", lvl, row+1, err)
			}
		}
		for col := 0; col < cols; col++ {
			if err := ctx.Err(); err != nil {
				return err
			}
			body, err := encodeDZITileFromStrips(prev, cur, next, cfg, lvl, lw, lh, col, row, quality)
			if err != nil {
				return fmt.Errorf("level %d (%d,%d): %w", lvl, col, row, err)
			}
			if err := w.WriteTile(lvl, col, row, body); err != nil {
				return err
			}
		}
		prev = cur
		cur = next
	}
	return nil
}
```

	// in cobra (which installs a SIGINT handler that cancels cmd.Context()).
	cmd := exec.Command(findBinary(t), "convert", "--to", "dzi", "-o", out, in)
	if err := cmd.Start(); err != nil {
		t.Fatalf("start: %v", err)
	}
	time.AfterFunc(100*time.Millisecond, func() {
		_ = cmd.Process.Signal(os.Interrupt)
	})
	_ = cmd.Wait()

	// Give worker goroutines a settle window.
	time.Sleep(200 * time.Millisecond)
	post := runtime.NumGoroutine()
	// Allow a small fudge for test-framework goroutines.
	if post > pre+5 {
		t.Errorf("goroutine leak: pre=%d post=%d", pre, post)
	}
}

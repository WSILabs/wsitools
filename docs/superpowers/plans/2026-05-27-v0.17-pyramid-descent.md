# wsitools v0.17 — Pyramid-Descent DZI/SZI + JPEG Codec Reorg Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make `convert --to dzi|szi` competitive with libvips `dzsave` (within 2× on CMU-1.ndpi) by replacing per-tile rendering with a single-iterator pyramid-descent generator + parallel JPEG encoder pool; concurrently reorganize the JPEG codec so vanilla YCbCr JPEG is the default and Aperio's APP14 quirk is preserved as a non-Factory direct type.

**Architecture:** One `*StripIterator` reads the source at L_max DZI dimensions; a chain of `levelBuilder`s downsamples 2× through every coarser DZI level via box filter, each holding a rolling 3-strip RGBA buffer for overlap. A parallel encoder pool (N=GOMAXPROCS) consumes tile jobs and pushes encoded bytes to a single sink-drain goroutine. The vanilla JPEG codec at `internal/codec/jpeg` is rewritten to emit standalone YCbCr+4:2:0 JPEGs without APP14; the previous Aperio variant moves verbatim to `internal/codec/aperioapp14` as a direct (non-Factory) type with zero current callers.

**Tech Stack:** Go 1.23+, opentile-go v0.26 (`*Slide.ScaledStrips`), libjpeg-turbo via cgo (`internal/codec/jpeg`'s existing C helper, modified to produce vanilla output), stdlib `image/png` for PNG fallback, `archive/zip` (consumed by `internal/szi`).

**Branch:** `feat/v17-pyramid-descent`.

**Spec:** `docs/superpowers/specs/2026-05-27-v0.17-pyramid-descent-design.md`.

```go
    "bytes"

func TestEncoderProducesAPP14Marker(t *testing.T) {
    enc, err := New(
        codec.LevelGeometry{Width: 16, Height: 16, TileWidth: 16, TileHeight: 16},
        codec.Quality{Q: 85},
    )
    if err != nil {
        t.Fatalf("New: %v", err)
    }
    defer enc.Close()

    // Construct a 16×16 RGB pattern (gradient).
    rgb := make([]byte, 16*16*3)
    for y := 0; y < 16; y++ {
        for x := 0; x < 16; x++ {
            i := y*16*3 + x*3
            rgb[i+0] = byte(x * 16)
            rgb[i+1] = byte(y * 16)
            rgb[i+2] = 0
        }
    }

    body, err := enc.EncodeTile(rgb, 16, 16, nil)
    if err != nil {
        t.Fatalf("EncodeTile: %v", err)
    }

    // Scan for APP14 marker (FF EE) within the abbreviated tile.
    // The marker should appear within the first ~30 bytes.
    found := false
    for i := 0; i+15 < len(body); i++ {
        if body[i] == 0xFF && body[i+1] == 0xEE {
            // Verify Adobe identifier "Adobe" at offset i+4..i+9.
            if bytes.Equal(body[i+4:i+9], []byte("Adobe")) {
                found = true
                break
            }
        }
    }
    if !found {
        t.Errorf("APP14 Adobe marker missing from encoded tile (len=%d)", len(body))
    }
}
```

```go
func TestEncodeStandaloneDecodesViaStdlib(t *testing.T) {
    enc, err := New(
        codec.LevelGeometry{Width: 64, Height: 64, TileWidth: 64, TileHeight: 64},
        codec.Quality{Q: 85},
    )
    if err != nil {
        t.Fatalf("New: %v", err)
    }
    defer enc.Close()

    rgb := makeRGBPattern(64, 64)
    body, err := enc.EncodeStandalone(rgb, 64, 64)
    if err != nil {
        t.Fatalf("EncodeStandalone: %v", err)
    }

    // Standalone JPEG must start with SOI (FF D8) and end with EOI (FF D9).
    if len(body) < 4 || body[0] != 0xFF || body[1] != 0xD8 {
        t.Fatalf("not a JPEG SOI prefix: %X", body[:4])

        t.Errorf("not a JPEG EOI suffix")

        if body[i] == 0xFF && body[i+1] == 0xEE {
            t.Errorf("vanilla output should not contain APP14 marker (offset %d)", i)
            break
        }
    }

    // Decode via stdlib image/jpeg and check approximate pixel match.
    img, err := stdjpeg.Decode(bytes.NewReader(body))
    if err != nil {
        t.Fatalf("stdlib decode: %v", err)

    if img.Bounds() != image.Rect(0, 0, 64, 64) {
```

```go
// Package jpeg provides a vanilla libjpeg-turbo-backed JPEG encoder.
// Output is YCbCr+4:2:0 chroma-subsampled JPEG with embedded
// DQT/DHT tables when EncodeStandalone is used, or abbreviated
// tiles (no tables; expected to be combined with LevelHeader via
// TIFF tag 347) when EncodeTile is used.
//

package jpeg

/*
#cgo pkg-config: libturbojpeg
#include <stdlib.h>
#include <string.h>
#include <jpeglib.h>

// wsi_encode_vanilla encodes a contiguous RGB buffer as JPEG via
// libjpeg-turbo. Output is YCbCr+4:2:0 (libjpeg's default) with no
// APP14 marker.
//
// abbreviated=0: full self-contained JPEG (SOI + DQT + DHT + SOS + scan + EOI).
// abbreviated=1: tile body only (SOI + SOS + scan + EOI). Tables must
//                be supplied separately (e.g. via TIFF tag 347).
//
// Caller frees *outBuf via free().
static int wsi_encode_vanilla(
    const unsigned char *rgb, int w, int h, int quality, int abbreviated,
    unsigned char **outBuf, unsigned long *outSize) {

    struct jpeg_compress_struct cinfo;
    struct jpeg_error_mgr jerr;
    cinfo.err = jpeg_std_error(&jerr);
    jpeg_create_compress(&cinfo);

    *outBuf = NULL;
    *outSize = 0;
    jpeg_mem_dest(&cinfo, outBuf, outSize);

    cinfo.image_width = w;
    cinfo.image_height = h;
    cinfo.input_components = 3;
    cinfo.in_color_space = JCS_RGB;          // input is RGB
    jpeg_set_defaults(&cinfo);                // → JCS_YCbCr output, 4:2:0
    jpeg_set_quality(&cinfo, quality, TRUE);

    if (abbreviated) {
        // Abbreviated stream: tables omitted from this output.
        // Caller writes DQT/DHT via TIFF tag 347.
        jpeg_suppress_tables(&cinfo, TRUE);
    }

    jpeg_start_compress(&cinfo, !abbreviated);  // write_all_tables=TRUE for standalone

    JSAMPROW row_pointer[1];
    int row_stride = w * 3;
    while (cinfo.next_scanline < cinfo.image_height) {
        row_pointer[0] = (JSAMPROW)(rgb + cinfo.next_scanline * row_stride);
        jpeg_write_scanlines(&cinfo, row_pointer, 1);
    }

    jpeg_finish_compress(&cinfo);
    jpeg_destroy_compress(&cinfo);
    return 0;
}

// wsi_compute_tables produces the JPEGTables blob (SOI+DQT+DHT+EOI)
// for the given quality. Used for TIFF tag 347 splice.
static int wsi_compute_tables(int quality, unsigned char **outBuf, unsigned long *outSize) {
    struct jpeg_compress_struct cinfo;
    struct jpeg_error_mgr jerr;
    cinfo.err = jpeg_std_error(&jerr);
    jpeg_create_compress(&cinfo);

    *outBuf = NULL;
    *outSize = 0;

    cinfo.image_width = 16;
    cinfo.image_height = 16;
    cinfo.input_components = 3;
    cinfo.in_color_space = JCS_RGB;
    jpeg_set_defaults(&cinfo);
    jpeg_set_quality(&cinfo, quality, TRUE);

    // Write tables only.
    jpeg_write_tables(&cinfo);
    jpeg_destroy_compress(&cinfo);
    return 0;
}

import "C"

import (
    "fmt"
    "runtime"
    "unsafe"

    "github.com/wsilabs/wsitools/internal/codec"
    "github.com/wsilabs/wsitools/internal/tiff"
)

func init() {
    codec.Register(Factory{})
}

type Factory struct{}

func (Factory) Name() string { return "jpeg" }

func (Factory) NewEncoder(g codec.LevelGeometry, q codec.Quality) (codec.Encoder, error) {
    return New(g, q)
}

// Encoder produces vanilla YCbCr+4:2:0 JPEGs.
type Encoder struct {
    geometry codec.LevelGeometry
    quality  int
    tables   []byte // precomputed JPEGTables for tag 347 splice
}

// New constructs a vanilla JPEG encoder.
func New(g codec.LevelGeometry, q codec.Quality) (*Encoder, error) {
    e := &Encoder{geometry: g, quality: q.Q}
    if e.quality < 1 || e.quality > 100 {
        e.quality = 85
    }
    if err := e.computeTables(); err != nil {
        return nil, err
    }
    return e, nil
}

func (e *Encoder) LevelHeader() []byte        { return e.tables }
func (e *Encoder) TIFFCompressionTag() uint16 { return tiff.CompressionJPEG }
func (e *Encoder) Close() error               { return nil }

func (e *Encoder) computeTables() error {
    var buf *C.uchar
    var size C.ulong
    if ret := C.wsi_compute_tables(C.int(e.quality), &buf, &size); ret != 0 {
        return fmt.Errorf("codec/jpeg: compute_tables failed")
    }
    defer C.free(unsafe.Pointer(buf))
    e.tables = C.GoBytes(unsafe.Pointer(buf), C.int(size))
    return nil
}

// EncodeTile encodes rgb as an abbreviated JPEG tile (no DQT/DHT
// tables). The returned bytes are a valid JPEG when combined with
// LevelHeader via TIFF tag 347 (JPEGTables) or by splicing the
// tables directly before SOS.
func (e *Encoder) EncodeTile(rgb []byte, w, h int, dst []byte) ([]byte, error) {
    out, err := e.encodeRaw(rgb, w, h, true)
    if err != nil {
        return nil, fmt.Errorf("codec/jpeg: EncodeTile: %w", err)
    }
    if dst != nil && cap(dst) >= len(out) {
        dst = dst[:len(out)]
        copy(dst, out)
        return dst, nil
    }
    return out, nil
}

// EncodeStandalone encodes rgb as a complete self-contained JPEG
// (SOI + DQT + DHT + SOS + scan + EOI). Used for DZI/SZI per-tile
// JPEGs where there is no level-shared tables tag.
func (e *Encoder) EncodeStandalone(rgb []byte, w, h int) ([]byte, error) {
    return e.encodeRaw(rgb, w, h, false)
}

func (e *Encoder) encodeRaw(rgb []byte, w, h int, abbreviated bool) ([]byte, error) {
    var outBuf *C.uchar
    var outSize C.ulong

    abbr := C.int(0)
    if abbreviated {
        abbr = 1
    }

    ret := C.wsi_encode_vanilla(
        (*C.uchar)(unsafe.Pointer(&rgb[0])),
        C.int(w), C.int(h),
        C.int(e.quality), abbr,
        &outBuf, &outSize,
    )
    runtime.KeepAlive(rgb)

    if ret != 0 {
        return nil, fmt.Errorf("codec/jpeg: encode failed (ret=%d)", int(ret))
    }
    defer C.free(unsafe.Pointer(outBuf))
    return C.GoBytes(unsafe.Pointer(outBuf), C.int(outSize)), nil
}
```

```go
// makeRGBA returns an image.RGBA of size w×h where pixel (x,y) =
// (id, byte(x), byte(y), 0xFF) — encodes strip id + column + row
// in R/G/B for traceable assertions.
func makeRGBA(w, h int, id byte) *image.RGBA {
    img := image.NewRGBA(image.Rect(0, 0, w, h))
    for y := 0; y < h; y++ {
        for x := 0; x < w; x++ {
            i := y*img.Stride + x*4
            img.Pix[i+0] = id
            img.Pix[i+1] = byte(x)
            img.Pix[i+2] = byte(y)
            img.Pix[i+3] = 0xFF
        }
    }
    return img
}
```

```go
package main

import (
    "image"

    "github.com/wsilabs/wsitools/internal/dzi"
)

// encodeJob hands a rendered tile RGBA to the encoder pool.
type encodeJob struct {
    level, col, row int
    img             *image.RGBA
}

// writeJob hands encoded bytes to the sink-drain goroutine.
type writeJob struct {
    level, col, row int
    body            []byte
}

// levelBuilder builds DZI tiles for one pyramid level. The chain
// of levelBuilders is linked via the `child` pointer (each level
// downsamples its strips into the next coarser level). The L_max
// builder has no parent; it receives strips from the ScaledStrips
// iterator directly. The coarsest builder (level=0) has child=nil.
type levelBuilder struct {
    level    int
    width    int  // pixel width at this level
    cfg      dzi.Config
    cols     int
    rows     int

    // Rolling 3-strip overlap buffer (each width × TileSize RGBA).
    prev, cur, next *image.RGBA

    // Accumulator for the strip currently being filled from
    // downsampled parent rows. nil for the L_max builder.
    accum     *image.RGBA
    accumRows int // rows of accum filled

    rowIndex int // next DZI row to emit

    child *levelBuilder
    jobs  chan<- encodeJob
}

// feed accepts one strip at this level's resolution. For L_max
// the strip comes from ScaledStrips; for coarser levels the strip
// is the result of accumulating downsampled rows from the parent.
//
// Behavior:
//   1. Rotate buffer: prev = cur; cur = next; next = strip.
//   2. If cur != nil, emit row rowIndex's DZI tiles.
//   3. Downsample strip 2× and feed to child (if any).
func (lb *levelBuilder) feed(strip *image.RGBA) {
    lb.prev = lb.cur
    lb.cur = lb.next
    lb.next = strip

    if lb.cur != nil {
        lb.emitRow(lb.rowIndex)
        lb.rowIndex++
    }

    if lb.child != nil {
        lb.child.acceptDownsampled(boxDownsample2x(strip))
    }
}

// acceptDownsampled is called by the parent with a half-height
// strip of this level's resolution. Accumulates into accum; when
// accum reaches TileSize rows, promotes it to a full strip and
// calls feed(accum).
func (lb *levelBuilder) acceptDownsampled(half *image.RGBA) {
    if lb.accum == nil {
        lb.accum = image.NewRGBA(image.Rect(0, 0, lb.width, lb.cfg.TileSize))
    }
    halfH := half.Bounds().Dy()
    halfW := half.Bounds().Dx()

    // Copy half's rows into accum starting at row lb.accumRows.
    for y := 0; y < halfH; y++ {
        if lb.accumRows+y >= lb.cfg.TileSize {
            break
        }
        srcRow := half.Pix[y*half.Stride : y*half.Stride+halfW*4]
        dstRow := lb.accum.Pix[(lb.accumRows+y)*lb.accum.Stride : (lb.accumRows+y)*lb.accum.Stride+halfW*4]
        copy(dstRow, srcRow)
    }
    lb.accumRows += halfH

    if lb.accumRows >= lb.cfg.TileSize {
        completed := lb.accum
        lb.accum = nil
        lb.accumRows = 0
        lb.feed(completed)
    }
}

// flush emits remaining DZI rows whose `cur`/`next` rolling buffer
// is incomplete. Called after the source iterator finishes.
func (lb *levelBuilder) flush() {
    // First, finalise any partial accumulator as a short strip.
    if lb.accum != nil && lb.accumRows > 0 {
        // Crop accum to its actual filled height.
        short := image.NewRGBA(image.Rect(0, 0, lb.width, lb.accumRows))
        for y := 0; y < lb.accumRows; y++ {
            copy(short.Pix[y*short.Stride:], lb.accum.Pix[y*lb.accum.Stride:y*lb.accum.Stride+lb.width*4])
        }
        lb.feed(short)
        lb.accum = nil
        lb.accumRows = 0
    }

    // Drain remaining rolling-buffer state. After the final feed,
    // cur and next exist but the last row hasn't been emitted yet.
    // We need one more rotation with next=nil to emit it.
    lb.prev = lb.cur
    lb.cur = lb.next
    lb.next = nil
    if lb.cur != nil && lb.rowIndex < lb.rows {
        lb.emitRow(lb.rowIndex)
        lb.rowIndex++
    }

    if lb.child != nil {
        lb.child.flush()
    }
}

// emitRow slices cur into the row's DZI tiles, with overlap from
// prev/next, and enqueues encodeJobs.
func (lb *levelBuilder) emitRow(row int) {
    for col := 0; col < lb.cols; col++ {
        tile := lb.assembleTile(col, row)
        lb.jobs <- encodeJob{level: lb.level, col: col, row: row, img: tile}
    }
}

// assembleTile builds the RGBA for DZI tile (col, row) by slicing
// from cur with top overlap from prev and bottom overlap from next.
func (lb *levelBuilder) assembleTile(col, row int) *image.RGBA {
    lw, lh := lb.width, lb.rows*lb.cfg.TileSize
    if lh > lb.width && row == lb.rows-1 {
        // last row may be short; compute actual content dim from cur.
    }
    cw, ch := dzi.EdgeTileDims(lw, lb.cfg.TileSize*lb.rows, lb.cfg.TileSize, col, row)
    _ = lh
    // The content height for the last short row equals cur's actual
    // pixel height. Recompute using cur:
    if lb.cur != nil && lb.cur.Bounds().Dy() < lb.cfg.TileSize {
        ch = lb.cur.Bounds().Dy()
    }

    leftOv := lb.cfg.Overlap
    if col == 0 {
        leftOv = 0
    }
    topOv := lb.cfg.Overlap
    if row == 0 || lb.prev == nil {
        topOv = 0
    }
    rightOv := lb.cfg.Overlap
    if col*lb.cfg.TileSize+cw >= lw {
        rightOv = 0
    }
    bottomOv := lb.cfg.Overlap
    if lb.next == nil {
        bottomOv = 0
    }

    outW := cw + leftOv + rightOv
    outH := ch + topOv + bottomOv

    dst := image.NewRGBA(image.Rect(0, 0, outW, outH))
    srcX0 := col*lb.cfg.TileSize - leftOv

    if topOv > 0 && lb.prev != nil {
        prevH := lb.prev.Bounds().Dy()
        copyRGBABand(dst, 0, lb.prev, prevH-topOv, srcX0, outW, topOv)
    }
    if lb.cur != nil {
        copyRGBABand(dst, topOv, lb.cur, 0, srcX0, outW, ch)
    }
    if bottomOv > 0 && lb.next != nil {
        copyRGBABand(dst, topOv+ch, lb.next, 0, srcX0, outW, bottomOv)
    }
    return dst
}

// copyRGBABand copies a horizontal band of pixels between two RGBAs.
// Out-of-range source columns are white-filled (defensive; the
// overlap suppression rules above should prevent this).
func copyRGBABand(dst *image.RGBA, dstY int, src *image.RGBA, srcY, srcX, w, h int) {
    srcW := src.Bounds().Dx()
    for r := 0; r < h; r++ {
        dstRowStart := (dstY + r) * dst.Stride
        srcRowStart := (srcY + r) * src.Stride
        for c := 0; c < w; c++ {
            sx := srcX + c
            var rr, gg, bb byte
            if sx >= 0 && sx < srcW {
                si := srcRowStart + sx*4
                rr = src.Pix[si+0]
                gg = src.Pix[si+1]
                bb = src.Pix[si+2]
            } else {
                rr, gg, bb = 0xFF, 0xFF, 0xFF
            }
            di := dstRowStart + c*4
            dst.Pix[di+0] = rr
            dst.Pix[di+1] = gg
            dst.Pix[di+2] = bb
            dst.Pix[di+3] = 0xFF
        }
    }
}

// boxDownsample2x reduces a source RGBA by 2x in both dimensions
// using 2x2 box-filter averaging. Used for DZI pyramid level
// generation between adjacent DZI levels.
func boxDownsample2x(src *image.RGBA) *image.RGBA {
    srcW := src.Bounds().Dx()
    srcH := src.Bounds().Dy()
    w := srcW / 2
    h := srcH / 2
    dst := image.NewRGBA(image.Rect(0, 0, w, h))
    for y := 0; y < h; y++ {
        for x := 0; x < w; x++ {
            sx, sy := x*2, y*2
            i0 := sy*src.Stride + sx*4
            i1 := sy*src.Stride + (sx+1)*4
            i2 := (sy+1)*src.Stride + sx*4
            i3 := (sy+1)*src.Stride + (sx+1)*4
            r := (uint32(src.Pix[i0+0]) + uint32(src.Pix[i1+0]) + uint32(src.Pix[i2+0]) + uint32(src.Pix[i3+0])) / 4
            g := (uint32(src.Pix[i0+1]) + uint32(src.Pix[i1+1]) + uint32(src.Pix[i2+1]) + uint32(src.Pix[i3+1])) / 4
            b := (uint32(src.Pix[i0+2]) + uint32(src.Pix[i1+2]) + uint32(src.Pix[i2+2]) + uint32(src.Pix[i3+2])) / 4
            di := y*dst.Stride + x*4
            dst.Pix[di+0] = byte(r)
            dst.Pix[di+1] = byte(g)
            dst.Pix[di+2] = byte(b)
            dst.Pix[di+3] = 0xFF
        }
    }
    return dst
}
```

```go
import (
    "context"
    "fmt"
    "image"
    bytespkg "bytes"
    "image/png"

    "github.com/wsilabs/wsitools/internal/codec/jpeg"
    "github.com/wsilabs/wsitools/internal/dzi"
)

// encoderWorker pulls encodeJobs, encodes via the supplied encoder,
// pushes writeJobs. Selects on ctx.Done() for cancellation.
func encoderWorker(ctx context.Context, jobs <-chan encodeJob, out chan<- writeJob,
    cfg dzi.Config, enc *jpeg.Encoder, quality int) {

    for {
        select {
        case <-ctx.Done():
            return
        case job, ok := <-jobs:
            if !ok {
                return
            }
            body, err := encodeTile(job.img, cfg.Format, enc, quality)
            if err != nil {
                // Encoder errors propagate via context cancel from caller side.
                return
            }
            select {
            case out <- writeJob{level: job.level, col: job.col, row: job.row, body: body}:
            case <-ctx.Done():
                return
            }
        }
    }
}

// encodeTile dispatches per cfg.Format. Returns encoded bytes.
func encodeTile(img *image.RGBA, format string, enc *jpeg.Encoder, quality int) ([]byte, error) {
    switch format {
    case "jpeg":
        // Convert RGBA → RGB for the encoder.
        w := img.Bounds().Dx()
        h := img.Bounds().Dy()
        rgb := make([]byte, w*h*3)
        for y := 0; y < h; y++ {
            for x := 0; x < w; x++ {
                si := y*img.Stride + x*4
                di := y*w*3 + x*3
                rgb[di+0] = img.Pix[si+0]
                rgb[di+1] = img.Pix[si+1]
                rgb[di+2] = img.Pix[si+2]
            }
        }
        return enc.EncodeStandalone(rgb, w, h)
    case "png":
        var buf bytespkg.Buffer
        if err := png.Encode(&buf, img); err != nil {
            return nil, err
        }
        return buf.Bytes(), nil
    default:
        return nil, fmt.Errorf("unsupported dzi format %q", format)
    }
}

// sinkDrainer pulls writeJobs and calls sink.WriteTile serially.
// Stores the first error in *firstErr (caller wraps with sync.Once).
func sinkDrainer(jobs <-chan writeJob, sink dziTileSink, firstErr *error) {
    for job := range jobs {
        if err := sink.WriteTile(job.level, job.col, job.row, job.body); err != nil {
            if *firstErr == nil {
                *firstErr = err
            }
        }
    }
}
```

```go
// runDescent drives the v0.17 pyramid-descent pipeline: one
// ScaledStrips iterator → linked levelBuilders → encoder pool →
// serialized sink drain. Returns the first error from any stage.
func runDescent(ctx context.Context, slide *opentile.Slide, sink dziTileSink, cfg dzi.Config, workers int) error {
    if workers <= 0 {
        workers = runtime.NumCPU()
    }

    ctx, cancel := context.WithCancel(ctx)
    defer cancel()

    quality := parseDZIQuality(cvQuality)

    // Build level chain top-down: top index = max, coarsest = 0.
    maxLevel := dzi.MaxLevel(cfg.Width, cfg.Height)
    encodeJobs := make(chan encodeJob, 2*workers)
    writeJobs := make(chan writeJob, 2*workers)

    var coarsest *levelBuilder
    var top *levelBuilder
    for lvl := 0; lvl <= maxLevel; lvl++ {
        lw, lh := dzi.LevelDims(cfg.Width, cfg.Height, lvl)
        cols, rows := dzi.GridDims(lw, lh, cfg.TileSize)
        lb := &levelBuilder{
            level: lvl, width: lw, cfg: cfg,
            cols: cols, rows: rows,
            jobs: encodeJobs,
        }
        if coarsest == nil {
            coarsest = lb
        } else {
            // The previous-iteration builder is one coarser; this one's
            // child IS that builder.
            lb.child = top // top currently holds the previous (coarser) iter's builder
        }
        top = lb
        _ = lh
    }
    // After the loop, `top` is the L_max builder; its `child` is
    // the next-coarser, and so on down to `coarsest`.

    // Encoder pool — one encoder per worker.
    encoders := make([]*jpeg.Encoder, workers)
    for i := 0; i < workers; i++ {
        enc, err := jpeg.New(
            codec.LevelGeometry{Width: cfg.TileSize, Height: cfg.TileSize, TileWidth: cfg.TileSize, TileHeight: cfg.TileSize},
            codec.Quality{Q: quality},
        )
        if err != nil {
            return fmt.Errorf("jpeg.New: %w", err)
        }
        encoders[i] = enc
        defer enc.Close()
    }

    var encWG sync.WaitGroup
    for i := 0; i < workers; i++ {
        encWG.Add(1)
        go func(idx int) {
            defer encWG.Done()
            encoderWorker(ctx, encodeJobs, writeJobs, cfg, encoders[idx], quality)
        }(i)
    }

    var sinkWG sync.WaitGroup
    var sinkErr error
    sinkWG.Add(1)
    go func() {
        defer sinkWG.Done()
        sinkDrainer(writeJobs, sink, &sinkErr)
    }()

    // Source iterator.
    stripOpts := []opentile.StripOption{opentile.WithStripContext(ctx)}
    if workers > 0 {
        stripOpts = append(stripOpts, opentile.WithStripWorkers(workers))
    }
    it := slide.ScaledStrips(
        image.Rect(0, 0, cfg.Width, cfg.Height),
        image.Point{X: cfg.Width, Y: cfg.Height},
        cfg.TileSize,
        stripOpts...,
    )
    defer it.Close()

    var srcErr error
    for {
        img, err := it.Next()
        if err == io.EOF {
            break
        }
        if err != nil {
            srcErr = err
            cancel()
            break
        }
        // Convert decoder.Image (RGB) to *image.RGBA.
        rgba := decoderImageToRGBA(img)
        top.feed(rgba)
    }
    if srcErr == nil {
        top.flush()
    }

    close(encodeJobs)
    encWG.Wait()
    close(writeJobs)
    sinkWG.Wait()

    if srcErr != nil {
        return srcErr
    }
    return sinkErr
}

// decoderImageToRGBA converts a decoder.Image (PixelFormatRGB) into
// *image.RGBA suitable for level builders.
func decoderImageToRGBA(img *decoder.Image) *image.RGBA {
    w, h := img.Width, img.Height
    dst := image.NewRGBA(image.Rect(0, 0, w, h))
    if img.Format == decoder.PixelFormatRGBA {
        for y := 0; y < h; y++ {
            copy(dst.Pix[y*dst.Stride:], img.Pix[y*img.Stride:y*img.Stride+w*4])
        }
        return dst
    }
    // RGB → RGBA.
    for y := 0; y < h; y++ {
        for x := 0; x < w; x++ {
            si := y*img.Stride + x*3
            di := y*dst.Stride + x*4
            dst.Pix[di+0] = img.Pix[si+0]
            dst.Pix[di+1] = img.Pix[si+1]
            dst.Pix[di+2] = img.Pix[si+2]
            dst.Pix[di+3] = 0xFF
        }
    }
    return dst
}
```

```bash
#!/usr/bin/env bash
# Compare wsitools convert --to dzi against vips dzsave on representative
# fixtures. Manual / not in CI. Run via `make bench-dzi`.

set -e
ROOT="$(cd "$(dirname "$0")/.." && pwd)"
WSI="$ROOT/bin/wsitools"
SAMPLES="${WSI_TOOLS_TESTDIR:-$HOME/GitHub/opentile-go/sample_files}"
TMP=$(mktemp -d)
trap "rm -rf $TMP" EXIT

if ! command -v vips >/dev/null; then
    echo "vips not in PATH; install libvips and re-run." >&2
    exit 1
fi
if [ ! -x "$WSI" ]; then
    echo "$WSI not built; run 'make build' first." >&2
    exit 1
fi

FIXTURES=(
    "$SAMPLES/svs/CMU-1-Small-Region.svs"
    "$SAMPLES/ndpi/CMU-1.ndpi"
    "$SAMPLES/ndpi/OS-2.ndpi"
)

run_timed() {
    local cmd="$1"
    local start end
    start=$(date +%s.%N)
    eval "$cmd" >/dev/null 2>&1
    end=$(date +%s.%N)
    echo "$end $start" | awk '{ printf "%.2f\n", $1 - $2 }'
}

printf "%-44s %12s %12s %8s\n" "fixture" "wsitools(s)" "vips(s)" "ratio"
printf -- "------------------------------------------------------------------------------\n"
for fx in "${FIXTURES[@]}"; do
    if [ ! -f "$fx" ]; then
        continue
    fi
    name=$(basename "$fx")

    rm -rf "$TMP/wsi-$name.dzi" "$TMP/wsi-${name}_files"
    t1=$(run_timed "'$WSI' convert --to dzi -o '$TMP/wsi-$name.dzi' '$fx'")

    rm -rf "$TMP/vips-$name.dzi" "$TMP/vips-${name}_files"
    t2=$(run_timed "vips dzsave '$fx' '$TMP/vips-$name' --suffix '.jpeg[Q=85]' --tile-size 256 --overlap 1")

    ratio=$(awk -v a="$t1" -v b="$t2" 'BEGIN{ if(b>0) printf "%.2fx", a/b; else print "n/a" }')
    printf "%-44s %12s %12s %8s\n" "$name" "$t1" "$t2" "$ratio"
done
```
